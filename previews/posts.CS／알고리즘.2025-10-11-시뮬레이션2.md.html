	while idx > 0:
		nxt = -1
		tmp = []
		for i in range(idx, -1, -1):
			if arr[i][j] == -1:
				nxt = i - 1
				break
			if arr[i][j] != 0:
				tmp.append(arr[i][j])
		
		i = idx
		while i >= 0:
			if arr[i][j] == -1:
				break
			arr[i][j] = 0
			i -= 1

		for i in range(len(tmp)):
			arr[idx - i][j] = tmp[i]
	
		idx = nxt
